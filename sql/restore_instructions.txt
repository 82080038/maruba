# Cara restore database:
# 1. Hapus database lama (jika perlu):
/opt/lampp/bin/mysql -u root -proot -e 'DROP DATABASE IF EXISTS maruba;'

# 2. Buat database baru:
/opt/lampp/bin/mysql -u root -proot -e 'CREATE DATABASE maruba;'

# 3. Restore dari dump:
/opt/lampp/bin/mysql -u root -proot maruba < /opt/lampp/htdocs/maruba/sql/maruba_database_dump_20260224_054409.sql

# 4. Verifikasi restore:
/opt/lampp/bin/mysql -u root -proot -e 'USE maruba; SHOW TABLES;'
